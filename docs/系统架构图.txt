  │                                                                               │

  │  BOOL[4]   上升沿 -> triggerStop(STOP1)     软急停                             │
  │  REAL[0]   -> plc_vision_offset_[0]        视觉偏移 X                         │
  │  REAL[1]   -> plc_vision_offset_[1]        视觉偏移 Y                         │
  │  REAL[2]   -> plc_vision_offset_[2]        视觉偏移 Z                         │
  │  REAL[3]   -> plc_vision_offset_[3]        视觉偏移姿态                        │
  │  BOOL[0~255] 全部 -> program_data_bool_[0~255]  供 .proc 程序引用              │
  │  INT[0~7]    全部 -> program_data_int_[0~7]     供 .proc 程序引用              │
  │  REAL[0~7]   全部 -> program_data_real_[0~7]    供 .proc 程序引用              │
  └───────────────────────────────────────────────────────────────────────────────┘

  ┌─ Robot -> PLC (buildRobotToPLCFrame 构建) ───────────────────────────────────┐
  │                                                                               │
  │  BOOL[0]   gpio_safety_get_state()==RUNNING   电源状态                         │
  │  BOOL[1]   EnableStatus==0                    告警                             │
  │  BOOL[2]   EnableStatus!=0                    已使能                           │
  │  BOOL[6]   trajectory_cache_.is_active        Robot_Busy (轨迹执行中)          │
